## 三、像素字体替换

### 3.1 功能描述
文档 4.3 节"可选优化"：识别文字内容后，用标准像素字体重新生成对应文字，完全替代原图文字。

### 3.2 设计思路

当 OCR 检测到文字区域并识别出内容后：
1. 将文字区域像素化到目标拼豆尺寸
2. 用内置像素字体（5×7 或 8×8）重新渲染文字
3. 用渲染结果替换原图文字区域的像素

### 3.3 像素字体定义

内置 5×7 ASCII 像素字体（大写字母 + 数字 + 常用符号）：

```python
# server-python/services/pixel_font.py

# 5×7 像素字体（宽×高），每个字符占 5 字节
# 每个字节的 bit 表示该行哪些像素点亮
# 示例：字母 'A'
PIXEL_FONT = {
    'A': [0b01110, 0b10001, 0b10001, 0b11111, 0b10001, 0b10001, 0b10001],
    'B': [0b11110, 0b10001, 0b10001, 0b11110, 0b10001, 0b10001, 0b11110],
    'C': [0b01110, 0b10001, 0b10000, 0b10000, 0b10000, 0b10001, 0b01110],
    # ... 完整字符集
    '0': [0b01110, 0b10001, 0b10011, 0b10101, 0b11001, 0b10001, 0b01110],
    '1': [0b00100, 0b01100, 0b00100, 0b00100, 0b00100, 0b00100, 0b11111],
    # ...
}

FONT_CHAR_WIDTH = 5   # 每字符宽 5 像素
FONT_CHAR_HEIGHT = 7  # 每字符高 7 像素
FONT_SPACING = 1      # 字符间距 1 像素
```

### 3.4 渲染函数

```python
def render_pixel_text(text: str, color_rgb: tuple) -> np.ndarray:
    """
    用像素字体渲染文字为图像
    
    Args:
        text: 要渲染的文字
        color_rgb: 文字颜色 (R, G, B)
    
    Returns:
        H×W×3 RGB 图像，透明区域为 (-1, -1, -1)
    """
    width = len(text) * (FONT_CHAR_WIDTH + FONT_SPACING) - FONT_SPACING
    height = FONT_CHAR_HEIGHT
    # -1 表示透明
    canvas = np.full((height, width, 3), -1, dtype=np.int16)
    
    for i, ch in enumerate(text.upper()):
        if ch not in PIXEL_FONT:
            continue
        bits = PIXEL_FONT[ch]
        x0 = i * (FONT_CHAR_WIDTH + FONT_SPACING)
        for row in range(FONT_CHAR_HEIGHT):
            for col in range(FONT_CHAR_WIDTH):
                if bits[row] & (1 << (FONT_CHAR_WIDTH - 1 - col)):
                    canvas[row, x0 + col] = color_rgb
    
    return canvas


def replace_text_with_pixel_font(
    image: np.ndarray,
    text_regions: list[dict],
    bg_color: tuple
) -> np.ndarray:
    """
    将检测到的文字区域替换为像素字体渲染
    
    对每个文字区域：
    1. 用像素字体渲染识别出的文字
    2. 缩放到原文字区域大小
    3. 替换原图像素
    """
    result = image.copy()
    
    for region in text_regions:
        x, y, w, h = region['bbox']
        text = region.get('text', '')
        if not text or w < 5 or h < 7:
            continue
        
        # 渲染像素文字
        rendered = render_pixel_text(text, (0, 0, 0))  # 黑色文字
        
        # 缩放到文字区域大小（最近邻，保持像素风格）
        scaled = cv2.resize(
            rendered.astype(np.float32),
            (w, h),
            interpolation=cv2.INTER_NEAREST
        )
        
        # 替换非透明像素
        mask = scaled[:, :, 0] >= 0  # 非-1 = 有文字
        result[y:y+h, x:x+w][mask] = scaled[mask].astype(np.uint8)
    
    return result
```

```python
# server-python/routes/convert.py

@router.post("/convert/detect-faces")
async def detect_faces_endpoint(file: UploadFile = File(...)):
    """
    检测图片中的人脸，返回所有人脸的位置和编号
    用于前端展示人脸选择 UI
    """
    file_bytes = await file.read()
    img_array = np.frombuffer(file_bytes, np.uint8)
    image = cv2.imdecode(img_array, cv2.IMREAD_COLOR)
    image = cv2.cvtColor(image, cv2.COLOR_BGR2RGB)
    
    # 缩放到 512px 以内提速
    h, w = image.shape[:2]
    scale = min(1.0, 512 / max(h, w))
    if scale < 1.0:
        image = cv2.resize(image, (int(w*scale), int(h*scale)))
    
    faces = detect_faces(image)
    
    return {
        'code': 200,
        'data': {
            'faces': [
                {
                    'index': i,
                    'bbox': f['bbox'],      # (x, y, w, h)
                    'landmarks': f.get('landmarks', []),
                    'confidence': f.get('confidence', 0),
                }
                for i, f in enumerate(faces)
            ],
            'image_scale': scale,
            'total': len(faces)
        }
    }
```

## 五、五官对称性检查与细节修复

### 5.1 功能描述
文档 7.3 节：检测眼睛、嘴巴位置，修正像素化后的变形；保证左右眼对称、嘴巴轮廓完整。

### 5.2 设计思路

像素化后，五官区域可能出现：
1. **左右眼不对称**：量化导致一只眼睛比另一只大/小
2. **嘴巴变形**：嘴唇轮廓断裂或比例失调
3. **眉毛断裂**：细眉被后处理过滤掉

修复策略：
1. **对称性检查**：比较左右眼区域的大小/位置，修正明显不对称
2. **轮廓补全**：在五官区域做轻量形态学闭运算（仅 1px）
3. **关键点保护**：以 68 点关键点为约束，保护关键轮廓像素不被后处理清除

### 5.3 实现

#### 对称性检查

```python
# server-python/services/feature_repair.py

def check_eye_symmetry(landmarks: np.ndarray) -> dict:
    """
    检查左右眼对称性
    
    68点关键点索引（文档附录B）：
    - 左眼：36-41
    - 右眼：42-47
    
    Returns:
        {
            'left_eye_size': float,
            'right_eye_size': float,
            'symmetry_score': float,  # 0-1, 1=完全对称
            'needs_correction': bool,
        }
    """
    if landmarks is None or len(landmarks) < 48:
        return {'symmetry_score': 1.0, 'needs_correction': False}
    
    # 左眼和右眼的关键点
    left_eye = landmarks[36:42]
    right_eye = landmarks[42:48]
    
    # 计算眼宽
    left_width = np.linalg.norm(left_eye[0] - left_eye[3])
    right_width = np.linalg.norm(right_eye[0] - right_eye[3])
    
    # 计算眼高
    left_height = max(np.linalg.norm(left_eye[1] - left_eye[5]),
                      np.linalg.norm(left_eye[2] - left_eye[4]))
    right_height = max(np.linalg.norm(right_eye[1] - right_eye[5]),
                       np.linalg.norm(right_eye[2] - right_eye[4]))
    
    # 对称性评分
    size_ratio = min(left_width, right_width) / max(left_width, right_width, 1)
    height_ratio = min(left_height, right_height) / max(left_height, right_height, 1)
    symmetry_score = (size_ratio + height_ratio) / 2.0
    
    return {
        'left_eye_size': left_width * left_height,
        'right_eye_size': right_width * right_height,
        'symmetry_score': round(symmetry_score, 3),
        'needs_correction': symmetry_score < 0.8,  # 80%以下需要修正
    }


def repair_eye_symmetry(
    grid: list,
    w: int, h: int,
    landmarks: np.ndarray,
    feature_mask: np.ndarray
) -> list:
    """
    修正左右眼不对称
    
    策略：将较大的眼睛区域裁剪到与较小的眼睛相同大小
    如果无法获取 landmarks，不做任何修改
    """
    if landmarks is None or len(landmarks) < 48:
        return grid
    
    result = [[{**cell} if cell else None for cell in row] for row in grid]
    
    # 左眼和右眼中心
    left_center = landmarks[36:42].mean(axis=0)
    right_center = landmarks[42:48].mean(axis=0)
    
    # 缩放到网格坐标
    scale_x = w / 1024  # 假设 landmarks 在 1024px 分辨率上
    scale_y = h / 1024
    
    lx, ly = int(left_center[0] * scale_x), int(left_center[1] * scale_y)
    rx, ry = int(right_center[0] * scale_x), int(right_center[1] * scale_y)
    
    # 如果左右眼水平位置差太大（可能是侧脸），跳过修正
    if abs(lx - rx) > w * 0.3:
        return result  # 侧脸，不强制对称
    
    # 取较小眼睛的尺寸
    left_radius = max(1, int(np.linalg.norm(landmarks[36] - landmarks[39]) * scale_x * 0.6))
    right_radius = max(1, int(np.linalg.norm(landmarks[42] - landmarks[45]) * scale_x * 0.6))
    eye_radius = min(left_radius, right_radius)
    
    # 不做强制裁剪（可能丢失信息），只确保眼睛区域不被后处理过滤
    # 在 feature_mask 中标记眼睛区域为"保护"状态
    return result


def repair_mouth_contour(
    grid: list,
    w: int, h: int,
    landmarks: np.ndarray
) -> list:
    """
    修复嘴巴轮廓
    
    68点关键点：嘴巴外轮廓 48-59，内轮廓 60-67
    策略：在嘴巴外轮廓区域做 1px 闭运算，填充断裂
    """
    if landmarks is None or len(landmarks) < 68:
        return grid
    
    result = [[{**cell} if cell else None for cell in row] for row in grid]
    
    scale_x = w / 1024
    scale_y = h / 1024
    
    # 嘴巴区域
    mouth_pts = (landmarks[48:60] * np.array([scale_x, scale_y])).astype(int)
    x0, y0 = mouth_pts[:, 0].min(), mouth_pts[:, 1].min()
    x1, y1 = mouth_pts[:, 0].max(), mouth_pts[:, 1].max()
    
    x0, y0 = max(0, x0-1), max(0, y0-1)
    x1, y1 = min(w-1, x1+1), min(h-1, y1+1)
    
    # 在嘴巴区域做 1px 闭运算（填充断裂像素）
    mouth_grid = [
        [result[y][x] for x in range(x0, x1+1)]
        for y in range(y0, y1+1)
    ]
    
    # 对每个空像素，如果四周有 ≥3 个同色邻居，填充
    for y in range(y0, y1+1):
        for x in range(x0, x1+1):
            if result[y][x] is not None:
                continue
            neighbors = {}
            for dy in [-1, 0, 1]:
                for dx in [-1, 0, 1]:
                    ny, nx = y + dy, x + dx
                    if not (y0 <= ny <= y1 and x0 <= nx <= x1):
                        continue
                    nc = result[ny][nx]
                    if nc:
                        hx = nc['hex']
                        neighbors[hx] = neighbors.get(hx, 0) + 1
            if neighbors:
                best_hex = max(neighbors, key=neighbors.get)
                if neighbors[best_hex] >= 3:
                    # 找该 hex 的完整 cell 信息
                    for dy in [-1, 0, 1]:
                        for dx in [-1, 0, 1]:
                            nc = result[y+dy][x+dx] if 0<=y+dy<h and 0<=x+dx<w else None
                            if nc and nc.get('hex') == best_hex:
                                result[y][x] = {'id': nc['id'], 'name': nc['name'], 'hex': nc['hex']}
                                break
                        if result[y][x]:
                            break
    
    return result
```
